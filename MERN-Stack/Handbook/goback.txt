import { useHistory } from "react-router-dom";
import "./App.css";
 
function App() {
  const history = useHistory();
 
  const handleGoBack = () => {
    history.goBack();
  };
  return (
    <>
      <button onClick={handleGoBack}>Go Back</button>
    </>
  );
}
 
export default App;
